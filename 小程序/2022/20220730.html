    TIP

    我们在开发小程序时，很多时候需要自定义 NavBar ，那么不可避免的要处理在不同设备下，页面展示正常，功能正常，这一篇将列举一些常见的兼容性问题

    # 不同设备下的页面展示

    小程序在不同设备下运行，可能会造成页面样式异常，下面针对 pc 端，安卓，苹果来实现兼容

    // 首页我们在 app.js 文件中写入以下代码

// app.js
App({
  onLaunch(e) {
    // 展示本地存储能力
     
  },
  globalData: {
    userInfo: null,
    visit_history:[],//品牌浏览历史
    is_iphone:'',//苹果刘海适配
    statusBarHeight:wx.getSystemInfoSync()['statusBarHeight'], 
    base_info:'',//全局基础信息，例如联系电话等
  }
})

// 调用 wx.getSystemInfoSync()['statusBarHeight'] 来获取当前设备的高度

// 第二步，在需要兼容的页面 js 文件中
// pages/addEnterprise/addEnterprise.js
const app = getApp() 引入
Page({
 data: { // statusBarHeight 判断是安卓还是苹果，20是安卓
        statusBarHeight: app.globalData.statusBarHeight,
     // platform：定义一个变量，判断是否是 pc 端，pc 是 windows  
     platform: '', 
 },
  onLoad(options) {
      //  wx.getSystemInfo 接口获取当前设备信息
        wx.getSystemInfo({
            success: (res) => {
                this.setData({
                    platform: res.platform
                })
            }
        })
    },
 })


// 示例使用
<!--pages/addEnterprise/addEnterprise.wxml-->
<view class="index">
    <view class="{{statusBarHeight == '20' ? 'header' : 'header-lh'}}" wx:if="{{platform !== 'windows'}}">
        <van-icon name="arrow-left" class="header-icon" size="50rpx" bindtap="handleback" />
        <text class="titles">加企微</text>
    </view>
</view>
