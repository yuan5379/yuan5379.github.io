    我们日常开发中，经常碰到给图片或者背景添加水印，这一篇以图片为例实现水印功能

    # 代码部分

    <!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta http-equiv="X-UA-Compatible" content="IE=edge" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
    <script src="./utils.js"></script>
    <style>
      * {
        margin: 0;
        padding: 0;
      }
    </style>
  </head>
  <body>
    <canvas id="img"></canvas>
    <script>
     
    function setWatermark (imgUrl, str) {
      var canvasImg = document.querySelector(`#img`);
      var ctxImg = canvasImg.getContext("2d"); // 获取画布canvas实例
      var img = new Image(); // 获取img实例
      img.src =imgUrl;  // img图片
      img.width = 500   // img宽度
      img.height = 400  // img高度
      canvasImg.width = img.width  // canvas宽度
      canvasImg.height = img.height // canvas高度
      // 图片加载完成触发事件
      img.onload = function(){
        // 将图片放到canvas上，参数是图片，x轴，y轴，图片宽度，图片高度
        ctxImg.drawImage(img, 0, 0, img.width, img.height); 

        // 创建水印 canvas
        var canvas = document.createElement('canvas');
        canvas.id = "canvas"
        // 设置canvas画布大小
        canvas.width = 80
        canvas.height = 30
    
        var ctx = canvas.getContext('2d')  // 获取水印canvas实例
        
        ctx.rotate(6.10); // 水印旋转角度
        // ctx.translate(0, 0);
        ctx.font = '14px Arial'
        ctx.fillStyle = 'skyblue'
        ctx.fillText(str, 0, 20) // 水印在画布的位置x，y轴
        
        // 计算铺满的数量循环
        var tamp_w = Math.ceil(canvasImg.width / canvas.width)
        var tamp_h = Math.ceil(canvasImg.height / canvas.height)+2

        for(var i = 0; i < tamp_h; i++) {
          for(var j = 0; j < tamp_w; j++) {
            canvasImg.getContext("2d").drawImage(canvas, j * canvas.width, i* canvas.height)
          }
        }
      }
    }
    setWatermark('https://www.runoob.com/images/img_the_scream.jpg','测试水印111111111111')
    </script>

</body>
