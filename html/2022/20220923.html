    TIP

    canvas 标签我们日常中会经常用到，这一篇用 canvas 标签实现一个简单的电子签名功能

    # 代码部分

    <!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta http-equiv="X-UA-Compatible" content="IE=edge" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
    <script src="./utils.js"></script>
    <style>
      * {
        margin: 0;
        padding: 0;
      }
      .cvs {
          border: 1px solid skyblue;
          position: relative;
          left: 700px;
          top: 40px;
      }
    </style>
  </head>
  <body>
    <canvas class="cvs" width="500" height="300"></canvas>
    <script>
       const cvs = document.querySelector('.cvs') // 获取 canvas 标签
       const ctx = cvs.getContext('2d') // 获取 canvas 对象，也就是渲染实例
       let isDrawing = false // 开关
       cvs.addEventListener('mousedown',e => {
           isDrawing = true
           ctx.moveTo(e.offsetX,e.offsetY) // 定位，初始化鼠标位置
       })
       cvs.addEventListener('mousemove',e => {
          if (isDrawing) {
            ctx.lineTo(e.offsetX,e.offsetY) // 绘画
            ctx.stroke()  // 渲染
          }
      })
      cvs.addEventListener('mouseup',e => {
          isDrawing = false
      })
      
    </script>

</body>
</html>
