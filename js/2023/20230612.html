    # 手写实现一个instanceof

    function myInstanceOf(left, right) {
  // 获取对象原型
  let proto = Object.getPrototypeOf(left);
  // 遍历原型链
  while (proto) {
    // 如果原型等于右侧构造函数的原型，返回 true
    if (proto === right.prototype) {
      return true;
    }

    // 继续向上查找原型链
    proto = Object.getPrototypeOf(proto);
  }

  // 如果没有找到，返回 false
  return false;
}

    # 手写Promise.all

    // 该函数接受一个包含 Promise 对象的数组参数，在所有 Promise 都成功的情况下，返回一个包含所有 Promise 结果的数组；如果其中一个 Promise 失败，则返回一个拒绝状态的 Promise，并将失败原因作为拒绝的原因。
function myPromiseAll(promises) {
  return new Promise((resolve, reject) => {
    const results = [];
    let completedPromisesCount = 0;

    promises.forEach((promise, index) => {
      Promise.resolve(promise)
        .then(result => {
          results[index] = result;
          completedPromisesCount++;

          if (completedPromisesCount === promises.length) {
            resolve(results);
          }
        })
        .catch(error => {
          reject(error);
        })
    });
  });
}

    # 手写一个并发函数

    /**
 * 
 * @param {string[]} urls 待请求的 url 数组
 * @param {number} maxNum 最大并发数 
 * @returns 
 */
function ComplicatedBy(urls,maxNum) {
  return new Promise((resolve) => {
      if (urls.length === 0) {
         resolve([])
         return
      }
      const results = [] // 存储成功的数组
      let index = 0 // 当前请求数组的下标
      let count = 0 // 当前请求完成的数量
      async function request() {
        if (index === urls.length) {
          return
        }
        const url = urls[index]
        const i = index // 记录放在 results 哪一个位置
        index++
        try {
           const resp = await fetch(url)
           results[i] = resp
        } catch (error) {
           results[i] = error
        } finally {
          count++
          if (count === urls.length) {
             resolve(results)
          }
          request()
        }
      }
      // 控制并发数
      for (let index = 0; index < maxNum; index++) {
        request()
      }
  })
}

          parentId: 5,
        },
  ];
 function convertTree(list) {
        const outlist = [];
        const map = {};
        for (let i = 0; i < list.length; i++) {
          const item = list[i];
          const parentId = item.parentId;
          if (!map[parentId]) {
             map[parentId] = [];
          } 
          map[parentId].push(item);
        }
        if (map[0]) {
          for (let i = 0; i < map[0].length; i++) {
            const item = map[0][i];
            item.children = map[item.id];
            outlist.push(item);
          }
        }
        return outlist;
      }
