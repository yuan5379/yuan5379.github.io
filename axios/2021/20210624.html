    import request from 'axios'
import qs from 'qs'
import helper from './helper'
import { Loading } from 'element-ui'
let loadingInstance
let loadingnum = 0
const showLoading = () => {
  loadingInstance = Loading.service({ fullscreen: true })
}
const hideLoading = () => {
  loadingInstance.close()
}
request.defaults.withCredentials = false
class Request {
  constructor () {
    // 公共参数
    request.defaults.params
    // 最大请求时间
    request.defaults.timeout = 120 * 1000
    // http code
    request.defaults.validateStatus = status => {
      return status >= 200 && status < 300
    }
    // 赋予内部
    this.instance = request
  }

  // 设置token
  setToken () {
    this.instance.defaults.headers.common['cmtoken'] = sessionStorage.getItem('login') ? JSON.parse(sessionStorage.getItem('login')).cmtoken : ''
  }

  /*
     * @apiGet 公共get请求
     * @apiPost 公共post请求
     * @apiJson post参数json格式
     * @apiFile 公共上传
     * @url 必填
     * @params 必填
     * */

  apiGet (url, params, config) {
    showLoading()
    loadingnum++
    return new Promise((resolve, reject) => {
      this.instance.get(`${url}`, {
        params,
        headers: {
          token: 'e7d07cc4378d1fe196bb7ceb02558e16bbbcdfd1ccc20a85731e5eabf1b09b26'
        },
        ...config
      }).then((res) => {
        loadingnum--
        if (!loadingnum) {
          hideLoading()
        }
        resolve(res.data)
      }).catch((error) => {
        loadingnum--
        if (!loadingnum) {
          hideLoading()
        }
        reject(error)
        helper.E('网络异常')
      })
    })
  }

  apiPost (url, params) {
    showLoading()
    return new Promise((resolve, reject) => {
      this.instance.post(`${url}`, qs.stringify(params)).then((res) => {
        hideLoading()
        resolve(res.data)
      }).catch((error) => {
        reject(error)
        hideLoading()
        helper.E('网络异常')
      })
    })
  }

  apiJson (url, params, config) {
    showLoading()
    loadingnum++
    this.setToken()
    return new Promise((resolve, reject) => {
      this.instance.post(`${url}`, params, { headers: { 'Content-Type': 'application/json' }, ...config }).then((res) => {
        loadingnum--
        if (!loadingnum) {
          hideLoading()
        }
        resolve(res.data)
      }).catch((error) => {
        loadingnum--
        if (!loadingnum) {
          hideLoading()
        }
        reject(error)
        helper.E('网络异常')
      })
    })
  }
  apiPut (params) {
    showLoading()
    // this.setToken()
    // let reader = new FileReader();
    // reader.readAsArrayBuffer(params.file); //转化二进制流，异步方法
    // reader.onload = (e) => { };
    return new Promise((resolve, reject) => {
      this.instance.put(params.url, params.file, { headers: { 'Content-Type': 'application/octet-stream;charset=utf-8' } }).then((res) => {
        hideLoading()
        resolve(res.data)
      }).catch((error) => {
        reject(error)
        hideLoading()
        helper.E('网络异常')
      })
    })
  }
  apiFile (url, fileKey, file) {
    showLoading()
    this.setToken()
    const formData = new FormData()
    formData.append(fileKey, file)

    return new Promise((resolve, reject) => {
      this.instance.post(url, formData, { headers: { 'Content-Type': 'multipart/form-data' } }).then((res) => {
        hideLoading()
        resolve(res.data)

        // this.checkResCode(res)
      }).catch(error => {
        hideLoading()
        reject(error)
        helper.E('网络异常')
      })
    })
  }
}

export default new Request()

    // 使用案例

import request from './request'
import { Message } from 'element-ui'

async function req ({
  url,
  method = 'post',
  params = {},
  config = {}
}) {
  const map = {
    post: request.apiJson,
    get: request.apiGet
  }

  if (url.startsWith('/customer') || url.startsWith('/file')) {
    url = process.env.VUE_APP_CM_URL + url // 0803分支上的原来设置 测试
    // url = 'https://cm.gewings.cn' + url // 客服生产
  } else if (url.startsWith('/ecard')) {
    url = process.env.VUE_APP_MP_URL + url
    // url = '' + url // 本地
  } else if (url.startsWith('/jchd')) {
    let newurl = url.split('/jchd')[1]
    url = process.env.VUE_APP_JC_URL + newurl
    // url = 'https://cloud-marketing-test-temp.oss-cn-shanghai.aliyuncs.com' + newurl // 本地
  } else {
    url = '/api-pla' + url
  }
  const res = await map[method].call(request,
    url,
    params,
    config
  )

  if (url.includes('/file/download')) {
    return res
  }

  if (url.includes('/ecard/file/')) {
    return res
  }
  if (url.includes('oss-cn')) {
    return res
  }

  const { errcode } = res
  // eslint-disable-next-line eqeqeq
  if (errcode == 0) {
    return res
  }
    
  Message.error(res.errmsg || '请求失败')
  throw new Error('request failed')
}

function reqPost (url, params, config) {
  return req({
    url,
    params,
    method: 'post',
    config
  })
}

function reqGet (url, params, config) {
  return req({
    url,
    params,
    method: 'get',
    config
  })
}

    # 原生小程序封装wx.request

    const handleFail = (config, errmsg) => {
    if (config.isShowToast) {
        wx.showToast({
            icon: 'none',
            title: errmsg || '请求失败',
        })
    }
    config && config.fail && config.fail(errmsg)
}

const BASE_URLS = {
    INFRA: 'https://bcard.erp.dahuahome.com' // 生产
    // INFRA: 'https://mingpian-test.gewings.cn' // 测试
}

export const request = (url, baseUrl = 'INFRA', method = 'get', data, config) => {
    wx.request({
        url: `${ BASE_URLS[baseUrl] + url}`,
        data,
        enableCache: true,
        enableHttp2: true,
        enableQuic: true,
        header: {
            token: '001449eae7d0b6df6afd6951b218bdf47c9523bb2ddebac053159a5216888e3b'
        },
        method,
        timeout: 300000,
        success: ({
            data
        }) => {
            if (data && data.errcode === 0) {
                config && config.success && config.success(data)
                return
            }
            config.isShowToast = true
            handleFail(config, data)
        },
        fail: ({
            errmsg
        }) => {
            config.isShowToast = true
            handleFail(config, errmsg)
        },
        complete: (res) => {},
    })
}
